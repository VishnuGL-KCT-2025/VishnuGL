#include <iostream>
#include <string>
#include <vector>
using namespace std;
int anagram(string s) {
    int len = s.length();

    
    if (len % 2 != 0) {
        return -1;
    }

   
    string s1 = s.substr(0, len / 2);
    string s2 = s.substr(len / 2);

    
    vector<int> count(26, 0); 

    for (char c : s1) {
        count[c - 'a']++;
    }

    for (char c : s2) {
        count[c - 'a']--;
    }

 
    int changes = 0;
    for (int i : count) {
        changes += abs(i);
    }

    return changes / 2;
}

    cin >> t;
